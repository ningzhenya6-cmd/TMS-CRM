## 二、WPS AirScript 脚本

在 WPS 表格中：
1. 点击菜单「工具」→「AirScript」
2. 新建脚本，粘贴以下代码
3. 保存，设置定时触发（建议每 10 分钟）

```javascript
/**
 * AcadAI 线索自动同步脚本
 * 从服务器拉取新线索，自动写入 WPS 表格
 * 
 * 使用方法：
 * 1. 打开这个 WPS 表格
 * 2. 工具 → AirScript → 粘贴此脚本
 * 3. 设置定时触发：每 10 分钟
 */

// ===== 配置 =====
const API_URL = 'http://39.96.8.59:8765/api/leads/list';
const SHEET_NAME = '线索管理';

// 列映射（API字段 → 表格列号 A=1, B=2, ...）
const COL_MAP = {
  '_createdAt': 1,    // A: 录入日期
  'name': 2,          // B: 学生称呼
  'wechat': 3,        // C: 微信号
  'phone': 4,         // D: 手机号
  'channel': 5,       // E: 来源渠道
  'sourceDetail': 6,  // F: 来源详情
  'countries': 7,     // G: 意向国家
  'school': 8,        // H: 学校
  'major': 9,         // I: 专业
  'degree': 10,       // J: 学历
  'demands': 11,      // K: 需求类型
  'urgency': 12,      // L: 紧急程度
  'status': 13,       // M: 跟进状态
  'recorder': 14,     // N: 负责人
  '_server_time': 15, // O: 最近跟进
  'note': 16,         // P: 跟进备注
};

/**
 * 主函数：同步线索
 */
function syncLeads() {
  try {
    // 1. 拉取数据
    const response = HTTP.get(API_URL);
    const data = JSON.parse(response.text());
    const leads = data.leads || [];
    
    if (leads.length === 0) {
      console.log('暂无新线索');
      return;
    }
    
    // 2. 获取当前工作表
    const sheet = Application.ActiveWorkbook.Sheets(SHEET_NAME);
    
    // 3. 找到最后一行
    let lastRow = sheet.UsedRange.Rows.Count;
    while (lastRow > 1 && !sheet.Cells(lastRow, 1).Value) {
      lastRow--;
    }
    
    // 4. 收集已有的 _server_time（去重用）
    const existingTimes = new Set();
    for (let r = 2; r <= lastRow; r++) {
      const timeVal = sheet.Cells(r, 15).Value;
      if (timeVal) existingTimes.add(String(timeVal));
    }
    
    // 5. 写入新数据
    let newCount = 0;
    for (let i = 0; i < leads.length; i++) {
      const lead = leads[i];
      const serverTime = lead._server_time || '';
      
      // 去重：已存在则跳过
      if (serverTime && existingTimes.has(serverTime)) continue;
      
      lastRow++;
      
      // 逐列写入
      for (const [field, colNum] of Object.entries(COL_MAP)) {
        const value = lead[field];
        const cell = sheet.Cells(lastRow, colNum);
        
        if (value === undefined || value === null) {
          cell.Value = '';
        } else if (Array.isArray(value)) {
          cell.Value = value.join('、');
        } else {
          cell.Value = value;
        }
      }
      
      // 新建行添加边框
      for (let c = 1; c <= 18; c++) {
        const cell = sheet.Cells(lastRow, c);
        cell.Borders.LineStyle = 1; // 细线边框
      }
      
      // 格式化日期列
      if (serverTime) {
        try {
          const cell = sheet.Cells(lastRow, 15);
          cell.NumberFormat = 'yyyy-mm-dd hh:mm';
        } catch(e) {}
      }
      
      newCount++;
    }
    
    console.log(`同步完成：新增 ${newCount} 条线索`);
    
  } catch (e) {
    console.log('同步失败：' + e.message);
  }
}

// 运行
syncLeads();
```

## 三、设置定时触发

1. WPS AirScript 编辑器 → 点击「触发器」
2. 添加触发器 → 选择函数 `syncLeads`
3. 触发方式：定时触发 → 每 10 分钟
4. 保存
